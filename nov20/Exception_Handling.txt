package nov20;


import java.util.Scanner;



public class Exceptional_Handling {
	public static void main(String[] args) {
		Scanner in = new Scanner(System.in);
		try {
			System.out.print("Enter an number for a:");
			int a = in.nextInt();
			System.out.print("Enter an number for b:");
			int b = in.nextInt();
			System.out.println("Result:"+(a/b));
			
			int[] arr= {1,2,3,4,5};
			System.out.println("Enter an index:");
			int n=in.nextInt();
			System.out.println(arr[n]);
			
			String name="janaki";
			System.out.println(name.charAt(3));
			
			String a1=null;
			System.out.println(a1.length());
		}
		catch(ArithmeticException e) {
			System.out.println("Try With Another Input!!!");
			System.out.print("Enter an number for a:");
			int a = in.nextInt();
			System.out.print("Enter an number for b:");
			int b = in.nextInt();
			System.out.println("Result:"+(a/b));
		}
		catch(ArrayIndexOutOfBoundsException e) {
			System.out.println("Please Enter Another Index Value!!!");
			int[] arr= {1,2,3,4,5};
			System.out.println("Enter an index:");
			int n=in.nextInt();
			System.out.println(arr[n]);
		}
		catch(StringIndexOutOfBoundsException e) {
			String name="janaki";
			System.out.println(name.charAt(3));
		}
		catch(NullPointerException e) {
			System.out.println("There is no values");
		}
		finally {
			System.out.println("Thank you!!!");
		}
	}

}
