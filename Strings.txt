package nov25;

public class Strings {
	
	public static void main(String[] args) {
		String s="Sathyabama";
		String s9="Sathyabama college";
		String s1=new String("Sathyabama");
		s.concat(" College");
		
		//upper&lower
		System.out.println(s.toUpperCase());
		System.out.println(s.toLowerCase());
		
		//trim
		System.out.println(s.trim());
		
		//start and end
		System.out.println(s.startsWith(" "));
		System.out.println(s.endsWith("a"));
		
		//index position
		System.out.println(s.charAt(3));
		
		//length
		System.out.println(s.length());
		
		//intern
		System.out.println(s.intern());
		
		//replace
		System.out.println(s.replace("S", "N"));
		
		//equals
		System.out.println(s.equals(s1));
		
		//==
		System.out.println(s==s9);
		
		//replaceAll
		System.out.println(s.replaceAll("a", "j"));
		
		//split
		System.out.println(s9.split(s9));
		
		//
		StringBuffer s0=new StringBuffer("Sathyabama");
		s0.append("College");
		StringBuilder s2=new StringBuilder("Sathyabama");
		s2.append("College");
		System.out.println(s0);
		System.out.println(s2);
	}
}
